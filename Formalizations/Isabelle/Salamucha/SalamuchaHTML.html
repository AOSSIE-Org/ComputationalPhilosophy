 541 lemma monotheism:
 542 assumes god: "(∃v. (¬ (f v) ∧ (∀u. (u ∈ (CC R) ∧ u ≠ v) ⟶ (R v u))))"  
 543 and c2: "K R"
 544 and c3: "(∃y. (y ∈ (CC R) ∧ (∀u. ((u ∈ (CC R) ∧ u ≠ y) ⟶ (R y u)))))"
 545 shows "((¬ (f v) ∧ (∀u. (u ∈ (CC R) ∧ u ≠ v) ⟶ (R v u))) ∧ (¬ (f w) ∧ (∀u. (u ∈ (CC R) ∧ u ≠ w) ⟶ (R w u))))
 546 ⟶ v = w"
 547 proof -
 548   {assume asm1: "(¬ (f v) ∧ (∀u. (u ∈ (CC R) ∧ u ≠ v) ⟶ (R v u)))"
 549    and asm2: "(¬ (f w) ∧ (∀u. (u ∈ (CC R) ∧ u ≠ w) ⟶ (R w u)))"
 550     {assume poly: "v ≠ w"
 551      from asm1 have v1: "∀x. ((x ∈ (CC R) ∧ (x ≠ v)) ⟶ (R v x))" by auto    
 552      from asm2 have w1: "∀x. ((x ∈ (CC R) ∧ (x ≠ w)) ⟶ (R w x))" by auto
 553       (*next step is not part of Salamucha´s outline*)     
 554      have vwin: "v ∈ (CC R) ∧ w ∈ (CC R)"
 555        proof - 
 556        from c3 obtain y where obty: "y ∈ (CC R)" by auto
 557        {assume "y ≠ v"   
 558          hence "v ∈ (CC R)" using v1 obty by auto}
 559        moreover
 560        {assume "y = v"
 561          hence "v ∈ (CC R)" using obty by simp}
 562        ultimately have "v ∈ (CC R)" by fastforce
 563        thus ?thesis using w1 obty by blast qed
 564      hence "(R v w) ∨ (R w v)" using c2 poly by blast  
 565     moreover
 566     {assume "R v w"
 567      hence "¬ (R w v)" using c2 by blast   
 568      hence False using w1 vwin poly by auto}
 569     moreover 
 570     {assume "R w v"
 571      hence "¬ (R v w)" using c2 by blast
 572      hence False using v1 vwin poly by auto}     
 573     ultimately have False by blast}
 574   hence "v = w" by blast}
 575 thus ?thesis by fast
 576 qed
